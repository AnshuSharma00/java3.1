package com.example.web;

import java.io.*;
import jakarta.servlet.*;
import jakarta.servlet.http.*;

public class LoginServlet extends HttpServlet {

    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws IOException, ServletException {

        response.setContentType("text/html");
        PrintWriter out = response.getWriter();

        String user = request.getParameter("username");
        String pass = request.getParameter("password");

        // Hardcoded validation (can be replaced with DB)
        if ("admin".equals(user) && "1234".equals(pass)) {
            out.println("<h2>Welcome, " + user + "!</h2>");
            out.println("<p>Login successful. Navigate to other modules:</p>");
            out.println("<ul>");
            out.println("<li><a href='employee.html'>View Employees</a></li>");
            out.println("<li><a href='attendance.jsp'>Student Attendance</a></li>");
            out.println("</ul>");
        } else {
            out.println("<h3 style='color:red;'>Invalid username or password</h3>");
            out.println("<a href='login.html'>Try Again</a>");
        }
    }
}
